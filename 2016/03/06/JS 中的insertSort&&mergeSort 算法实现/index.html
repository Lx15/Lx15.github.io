最近在学习算法，所以将刚学的插入排序和归并排序用js实现一下。

insertSort

function insertSort(arr){
    var l=arr.length;
    for(var j=1;j < l ;j++){
        var key = arr[j];
         var i = j-1;
        while(  i>=0 && arr[i] > key){
             arr[i+1]=arr[i];
             i--;
        }
        arr[i+1] =key;
    }
}

  • 注释：注意哨兵key的使用

    mergeSort

    function merge(arr,p,q,r){

     var n1 = q-p+1;
 var n2 = r-q;
 var L=[],R=[];
 for(var i = 0;i < n1;i++){
   L[i] = arr[p+i];
  }
for(var j = 0;j < n2;j++){
   R[j] = arr[q+j+1];
}
 L[n1] = Infinity;
 R[n2] = Infinity;
 var i=0,j=0;
 for(var k=p;k<= r;k++){     
    if(L[i] <= R[j]){     
      arr[k] = L[i];
      i++;
    }else{
      arr[k] = R[j];
      j++;
   }

    }
    function mergeSort(arr,p,r){

    if(p < r){
  // q=Number.parseInt((p+r)/2);
  //.........
  var q=Math.floor((p+r)/2); 
  mergeSort(arr,p,q);         }

  mergeSort(arr,q+1,r);
  merge(arr,p,q,r);

    }

  • 在mergeSort中最需要注意的是函数中的局部作用域，在最初的代码中mergeSort 中的q,因为没有写var而变成全局变量，所以出错。
